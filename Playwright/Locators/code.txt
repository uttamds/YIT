    import { test, expect } from '@playwright/test';

test('Demo various Playwright locators', async ({ page }) => {

  await page.goto('file:///C:/path-to-your/locators-demo.html'); 
  // Path doesn't matter since you're only demonstrating

  // 1Ô∏è‚É£ getByText ‚Üí Finds element by visible text
  await expect(page.getByText('Welcome to Playwright Training')).toBeVisible();

  // 2Ô∏è‚É£ getByRole ‚Üí Based on ARIA role
  await page.getByRole('button', { name: 'Submit' }).click();

  // 3Ô∏è‚É£ getByLabel ‚Üí Input associated with label
  await page.getByLabel('Full Name').fill('Amit Kumar');

  // 4Ô∏è‚É£ getByPlaceholder ‚Üí Using placeholder attribute
  await page.getByPlaceholder('Enter your email').fill('amit@email.com');

  // 5Ô∏è‚É£ getByAltText ‚Üí For images
  await expect(page.getByAltText('Student Image')).toBeVisible();

  // 6Ô∏è‚É£ getByTitle ‚Üí Title attribute
  await page.getByTitle('Close Form').click();

  // 7Ô∏è‚É£ CSS Selector ‚Üí Class
  await expect(page.locator('.price')).toHaveText('$99');

  // 8Ô∏è‚É£ CSS Selector ‚Üí ID
  await expect(page.locator('#main-title')).toHaveText('Student Registration');

  // 9Ô∏è‚É£ XPath Locator
  await expect(page.locator('//li[text()="Python"]')).toBeVisible();

  // üîü Chained Locator
  await expect(page.locator('.card').locator('.price')).toHaveText('$99');

});
